	ORG 00H
	JMP MAIN
	ORG 50H
MAIN:
	MOV		TMOD,#0			; Mode 0, Timer
	MOV		TL0,#0	
	MOV		TH0,#131
	SETB	TR0
	MOV 	A,#11000000B
START:
	MOV 	P0,A
	MOV 	R4,#250
DELAY1S:
	ACALL 	DELAY
	DJNZ	R4,DELAY1S
	RR		A
	SJMP	START
;4 ms
DELAY:
	JNB		TF0,$		; Timer 1 overflow flag
	CLR		TF0
	MOV 	TL0,#0
	MOV		TH0,#131
	RET	
END
